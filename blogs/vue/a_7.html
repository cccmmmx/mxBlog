7、call和apply、bind

//call和apply、bind可以执行函数并改变this的指向
--------------------------------------------------------
//call使用
const obj={
    text:3
}
function fn(a,b){
    console.log(a+b+this.text)
}
//将fn的this指向obj并且执行，并且已单个数的方式传参
fn.call(obj,1,2) //6




//call的经典用法 实现继承
function Father(name,age){
    this.name = name
    this.age = age
}
function Son(name,age){
    Father.call(this,name,age)
}
const son = new Son('新哥帅',20)
console.log(son)//Son继承Father中的属性
-------------------------------------------------------------

//apply的用法和call差不多，不够有个最大的区别就是传参是数组的形式
const obj={
    text:3
}
function fn(a,b){
    console.log(a+b+this.text)
}
//将fn的this指向obj并且执行，并且已单个数的方式传参
fn.apply(obj,[1,2]) //6




//apply常用，比如求最大值
const arr = [1,2,3,4,5,6]
const max = Math.max.apply(Math,arr)
console.log(arr)  //6
//当然，这个方法看个人使用，个人认为 Math.max(...arr) 更方便

//bind不会执行函数，只会改变this指向并返回新的函数
const obj={
    text:3
}
function fn(a,b){
    console.log(this.text)
}
var f = fn.bind(obj)
 f() //3

# 三、严格模式

js除了正常模式外，还有严格模式。严格模式是ES5才提出，支持IE10以上。

严格模式顾名思义，它是代码更为严格一些，让我们写代码不能太过自由

  • 消除了 Javascript 语法的一些不合理、不严谨之处，减少了一些怪异行为。
  • 消除代码运行的一些不安全之处，保证代码运行的安全。 提高编译器效率，增加运行速度。
  • 禁用了在 ECMAScript 的未来版本中可能会定义的一些语法，
  • 为未来新版本的 Javascript 做好铺垫。比如一些保留字如：class,enum,export, extends, import, super 不能做变量名

严格模式分为两种，为脚本开启严格模式和为函数开启

  • 为脚本开启严格模式

    <script>
    //开启严格模式很简单，在<script>脚本里面最上面添加"use strict"字符串就可以
  　"use strict"; //开启了严格模式
    
    //在严格模式下，不能使用未定义的变量
    text = '123'
    console.log(text)//报错
    
    //不能随便删除声明的变量
    var text='123'
    delete text; //报错
    
    //全局作用域中的函数this是undefined
    function fn(){
        console.log(this)
    }
    fn() //undefined
    
    //不能在非函数代码块中声明函数
    //1.比如在if代码块里 报错
    if(true){
        function fn(){}
    }
    //2.在函数代码块里声明函数 可以
    function myFn(){
        //严格模式下函数代码块里是可以声明函数的
        function fn(){
        }
    }
</script>

# 四、高级函数

高级函数一听高阶两个字就怕，其实也还好，不是特别难理解。高级函数是指A函数接收的参数是一个B函数或者A函数返回值是一个B函数，就称A函数是一个高级函数，闭包就是一个典型的高级函数

# 五、闭包

​ 记得之前出去面试，闭包几乎是所有面试官的喜好，感觉就像是不问闭包就不是一个合格的面试官一样。不过可以理解，闭包算是js的高级使用，而且在let没有出来的之前，闭包解决了许多变量访问的问题，用法也是非常广泛，像回调函数之类。总之，闭包还是要搞懂和使用。

​ 闭包的定义是这样说的，闭包是指有权访问另一个函数作用域中变量的函数。简单来说就是一个函数可以去访问另一个函数的局部作用域。

//闭包简单案例
function fn(){
    var num =10;
    function myfn(){
        console.log(num)
    }
    myfn()
}
fn()
//打印 10，可以看到myfn访问了fn中的变量，fn作为被访问的函数，就是闭包函数

可能你会觉得上面的案例平平无奇，这不是正常操作吗，就是这么个正常操作，让我们可以在fn的外部访问到num，我们改造一下

function fn(){
    var num =10;
    function myfn(){
       console.log(num)
    }
   return myfn
}


fn()()  //打印 10


var f = fn()
f() //打印 10
//这样就成功实现在fn外面的作用域访问到fn中的num
//你可能会觉得，直接返回不是更完美
//像这样
function fn(){
    var num=10
    return num
}
//这种确实可以获取到变量，但是不同的是这种方法在函数被调用后，变量就会被垃圾回收机制清除，如果我们要一直持续访问，就需要利用闭包的原理。就像上面闭包的例子，利用myfn函数来保存fn闭包函数中的局部变量，使fn在执行后，局部变量num一直存在且可以访问。

//在let没有出来之前，闭包在面试题中有个经典的题目
for(var i=0;i<3;i++){
    setTimeout(function(){
        console.log(i)
    },i*1000)
} //打印555，之所以打印555是因为在执行for循环结束后，才开始执行setTimeout延时的异步操作，因为循环已经结束，所以此时i是5，所以执行555
//执行过程如下，i相当于是for循环里的全局变量
//第一次循环i=0,此时还没有打印
 setTimeout(function(){
        console.log(i)
   },i*1000)
//第二次循环i=1，此时还没有打印
  setTimeout(function(){
        console.log(i)
   },i*1000)
//第三次i=2，此时还没有打印
  setTimeout(function(){
        console.log(i)
   },i*1000)
//第四循环i=3，循环结束，此时i=3，所以打印都是3


//利用闭包
for(var i=0;i<3;i++){
   (function(i){
      setTimeout(function(){
        console.log(i)
       },i*1000) 
    } 
    )(i)
} //打印012，之所以不一样是每次for循环的时候，匿名闭包函数下面的子函数，每次都执行一个新的堆内存保存i的值，所以在执行setTimeout异步时，i相当于不同的局部作用域被保存起来，这也是为什么可以打印不同的i
//执行过程如下 每次循环i是匿名函数中的局部变量
//第一次循环i=0
(function(0){
      setTimeout(function(){
        console.log(0)
       },0*1000) 
    } )(0)
//第一次循环i=1
(function(1){
      setTimeout(function(){
        console.log(1)
       },1*1000) 
    } )(1)
//第一次循环i=2
(function(2){
      setTimeout(function(){
        console.log(2)
       },2*1000) 
    } )(2)
//记住在每次循环匿名函数都会立即执行。可能我表达的不是很好，所以一定要好好理解一下。

​ 在实际开发中，我们不可能把所有的变量都定义为全局作用域，这样也是为了避免重名变量之间的干扰，而且全局变量往往会带来后期维护中的许多问题。但是如果有要定义局部变量，又要在其他地方访问，闭包就是个不错的选择。

​ 不过，闭包也要避免常用，通过闭包函数来访问局部变量，都会重新开辟一个新的堆内存，大量使用闭包也可能会造成内存泄漏，所以要根据具体情况，慎用闭包。

总结一下闭包功能

1、扩展函数局部变量的操作和访问范围

2、使函数的局部变量不会被垃圾回收机制清除，一直存在内存中

# 六、函数递归

函数递归，指的是一个函数在内部调用自身，就是函数递归，简称函数套娃。函数递归简直就是“栈溢出”的知音，如果没有条件return结束递归，就会一直调用自己，直到栈溢出错误，浏览器卡死。

var num=0
function fn(){
    console.log(num)
    if(num==6){
        return false;
    }
    num++
    fn()
}
fn() //打印0123456

//求阶乘1*2*3*4*5*6*...n
function fn(n){
    if(n==1){
        return 1
    }
    return n*fn(n-1)
}
console.log(fn(3))//6


//求斐波那契数列(斐波那契数列就是除1和2外，第n项是第n-1和n-2之和，像1,2,3,5，8,13，21...n-2,n-1,n)
function fb( n ){
	if(n == 1 || n == 2){  //递归出口
		return 1 ;
	}else{
		return fb( n - 1 ) + fb( n - 2 ); //递归体
	}		
}

# 六、浅拷贝与深拷贝

​ 在看浅拷贝和深拷贝之前，一定要理解栈内存和堆内存以及普通类型和对象是如果保存在内存中的，像Number和String这些是保存在栈内存中，而Object类型，像对象先是在栈内存中保存一个地址，该地址指向堆内存，所以对象的赋值就是地址的赋值，看下面地址。

var obj1={a:1,b:1}
var obj2=obj1
obj2.a=2
console.log(obj1.a) //2
console.log(obj2.a) //2

​ 其实不难理解，obj1赋值给obj2，相当于将obj1在栈中的地址赋值给obj2，此时obj2的地址也是指向obj1中的堆内存，也就是obj1和obj2指向相同的堆内存，所以改变obj2中a的值为2后，堆内存中的a=2，obj1此时去取a就是2。也就是说，改变obj2也是在改obj1。

​ 所以对象的赋值要避免这种情况，赋值不同的变量互相干扰，所以就需要浅拷贝，将上面的对象进行浅拷贝

var obj1={a:1,b:1}

//普通做法，简单理解与直接
var obj2={} //新的栈地址指向新的堆内存
obj2.a=obj1.a
obj2.b=obj1.b

//或者对象解构赋值
var obj2={...obj1} //解构赋值会产生新的对象
obj2.a=2

console.log(obj1.a) //1
console.log(obj2.a) //2

​ 可以看到obj下面有a和d属性，a属性是一个对象，有b属性，b属性也是一个对象，有c属性，如果我们遇到这种多层的对象，用简单的浅拷贝给obj2，那么在obj2改变b的值时，obj中b的值也会改变，所以我们再拷贝的时候，也需要将对象里面属性是对象进行拷贝，也就是深拷贝

//深拷贝有很多种，我个人用得较多的就两种

//第一种,如果要拷贝的对象中属性只是普通的类型和对象，没有函数或者正则表达式的话，就可以用下面的方法
var obj={
    a:{
        b:{
            c:1
        }
    },
    d:2
}
//将对象转成JSON字符串，此时obj变成字符串保存在栈中，JSON.parse将JSON字符串转成对象，产生一个新的地址，也就是新的对象，里面的对象属性b和c也是一样的道理，但是这种局限性比较多，用于简单对象的深拷贝
var obj2=JSON.parse(JSON.stringify(obj))

//第二种 通过递归的方式，将对象以及里面的对象属性重新定义和赋值
//经典用法 深拷贝
function deepClone(origin) {
    //判断属性是对象还是null
	var isObject = any => typeof any == 'object' && any != null
    //判断属性是否是数组
	var isArray = any => Object.prototype.toString.call(any) === '[object Array]'
    //如果不是对象，直接返回
	if (!isObject(origin)) return origin
    //判断是数组还是对象，如果是数组，通过字面量[]定义新的数组，是对象就通过{}定义新对象
	var target = isArray(origin) ? [] : {}
	for (var prop in origin) {
         //判断属性是否是原有的，不是原型链上面的
		if (origin.hasOwnProperty(prop)) {
			var value = origin[prop]
			if (isObject(value)) {
				target[prop] = deepClone(value)
			} else {
				target[prop] = value
			}
		}
	}
	return target
}
